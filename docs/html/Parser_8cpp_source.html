20/*
21 * Function: parseIntList
22 * ----------------------
23 * Parses a list of integers inside square brackets or separated by commas/semicolons.
24 *
25 * Example:
26 * "[1, 2, 3]" → {1, 2, 3}
27 */
28std::vector<int> parseIntList(const std::string& inside) {
29 std::vector<int> vals;
30 if (inside.empty()) return vals;
31
32 std::string t = inside;
33 for (char& c : t)
34 if (c == ',' || c == ';') c = ' ';
35
36 std::stringstream ss(t);
37 std::string token;
38 while (ss >> token) {
39 try {
40 size_t pos = 0;
41 int v = std::stoi(token, &pos);
42 if (pos != token.size()) return {}; // not fully numeric
43 vals.push_back(v);
44 }
45 catch (...) {
46 return {}; // contains non-numeric
47 }
48 }
49 return vals;
50}

53 * Function: parse_record_line
54 * ---------------------------
55 * Parses a record definition line from the input file.
56 *
57 * Example:

62bool parse_record_line(const std::string& line, Record& rec, std::set<Error>& errors) {
63 if (line.empty()) {
64 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
65 errors.insert(e);
66 return false;
67 }
68
69 size_t colon = line.find(':');
70 if (colon == std::string::npos) {
71 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
72 errors.insert(e);
73 return false;
74 }
75
76 rec.name = trim(line.substr(0, colon));
77 if (rec.name.empty()) {
78 Error e{ ErrorCode::EMPTY_RECORD_NAME, "Parser" };
79 errors.insert(e);
80 return false;
81 }
82
83 std::string propsPart = trim(line.substr(colon + 1));
84 if (propsPart.empty()) {
85 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
86 errors.insert(e);
87 return false;
88 }
89
90 std::vector<std::string> chunks;
91 std::string cur;
92 int depth = 0;
93
94 // split properties by commas, respecting brackets
95 for (char c : propsPart) {
96 if (c == '[') depth++;
97 if (c == ']') depth--;
98 if (c == ',' && depth == 0) {
99 if (!cur.empty()) {
100 chunks.push_back(trim(cur));
101 cur.clear();
102 }
103 }
104 else {
105 cur.push_back(c);
106 }
107 }
108 if (!cur.empty()) chunks.push_back(trim(cur));
109 if (chunks.empty()) {
110 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
111 errors.insert(e);
112 return false;
113 }
114
115 for (auto& token : chunks) {
116 if (token.empty()) continue;
117
118 size_t eq = token.find('=');
119 if (eq == std::string::npos) {
120 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
121 errors.insert(e);
122 return false;
123 }
124
125 std::string pname = trim(token.substr(0, eq));
126 std::transform(pname.begin(), pname.end(), pname.begin(),
127 [](unsigned char c) { return std::tolower(c); });
128
129 std::string val = trim(token.substr(eq + 1));
130 if (val.size() < 2 || val.front() != '[' || val.back() != ']') {
131 Error e{ ErrorCode::INCORRECT_RULE, "Parser" };
132 errors.insert(e);
133 return false;
134 }
135
136 std::string inside = val.substr(1, val.size() - 2);
137 Property p{ pname, parseIntList(inside) };
138
139 // reject duplicate property
140 if (rec.properties.count(pname)) {
141 Error e{ ErrorCode::DUPLICATE_PROPERTY, "Parser" };
142 errors.insert(e);
143 return false;
144 }
145
146 // reject non-numeric values
147 if (!inside.empty() && p.values.empty()) {
148 Error e{ ErrorCode::INVALID_NUMERIC_VALUE, "Parser" };
149 errors.insert(e);
150 return false;
151 }
152
153 rec.properties[pname] = p;
154 }
155
156 if (rec.properties.empty()) {
157 Error e{ ErrorCode::INVALID_RECORD, "Parser" };
158 errors.insert(e);
159 return false;
160 }
161
162 return true;
163}
